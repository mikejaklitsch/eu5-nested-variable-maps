on_game_start = {
	on_actions = { nvm_run_tests_startup }
}

weather_monthly_pulse = {
	on_actions = { nvm_run_tests_monthly }
}

nvm_run_tests_startup = {
	effect = {
		error_log = "NVM === game_start ==="
		nvm_init_test_pools = yes
		nvm_run_tests_impl = yes
		error_log = "NVM === stress ==="
		nvm_run_stress_tests = yes
	}
}

nvm_run_tests_monthly = {
	trigger = { NOT = { has_global_variable = nvm_monthly_done } }
	effect = {
		set_global_variable = nvm_monthly_done
		error_log = "NVM === monthly_pulse ==="
		nvm_init_test_pools = yes
		nvm_run_tests_impl = yes
	}
}
