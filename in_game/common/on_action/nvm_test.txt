# All output goes to logs/error.log. Grep for prefix to filter:
#   NVM       core + stress tests
#   NVM_CAP   engine capability probes
#   NVM_TUT   tutorial (horse racing)
#   NVM_PAR   parallel safety tests
on_game_start = {
	on_actions = {
		nvm_run_tests_startup
		nvm_run_capability_tests_action
		nvm_run_tutorial_action
		nvm_parallel_setup_action
	}
}

# ═══════════════════════════════════════════════════════════════════════════════
# PARALLEL SAFETY TESTS
# ═══════════════════════════════════════════════════════════════════════════════
# Uncomment the monthly_country_pulse block below to run parallel safety tests.
# These use events = { } dispatch for engine-level parallelization by country.
#
# Structure:
#   1. weather_monthly_pulse fires nvm_parallel_setup (sequential, on_actions)
#   2. monthly_country_pulse fires nvm_parallel_dispatch (parallel, events)
#   3. Next weather_monthly_pulse fires nvm_parallel_verify (sequential)
#
# Grep for NVM_PAR in logs/error.log to filter output.
#
monthly_country_pulse = {
	on_actions = { nvm_parallel_dispatch }
}

weather_monthly_pulse = {
	on_actions = {
		nvm_parallel_verify_action
		nvm_parallel_clobber_action
	}
}

# ═══════════════════════════════════════════════════════════════════════════════
# CORE TEST DISPATCH
# ═══════════════════════════════════════════════════════════════════════════════

nvm_run_tests_startup = {
	effect = {
		debug_log = "NVM === game_start ==="
		nvm_run_tests_implementation = yes
		debug_log = "NVM === stress ==="
		nvm_run_stress_tests = yes
	}
}

nvm_run_tests_monthly = {
	trigger = { NOT = { has_global_variable = nvm_monthly_done } }
	effect = {
		set_global_variable = nvm_monthly_done
		debug_log = "NVM === monthly_pulse ==="
		nvm_run_tests_implementation = yes
	}
}

# 200k stress test. Opt-in only: extremely resource-heavy, creates 200,000
# dead characters. Uncomment the weather_monthly_pulse hook above to enable.
nvm_run_absurd_stress = {
	trigger = {
		has_global_variable = nvm_monthly_done
	}
	effect = {
		nvm_run_absurd_stress_impl = yes
	}
}

# ═══════════════════════════════════════════════════════════════════════════════
# PARALLEL TEST ON_ACTIONS
# ═══════════════════════════════════════════════════════════════════════════════

nvm_parallel_setup_action = {
	effect = {
		nvm_parallel_setup_impl = yes
	}
}

nvm_parallel_dispatch = {
	events = {
		nvm_parallel_test.1
		nvm_parallel_test.2
		nvm_parallel_test.3
	}
}

nvm_parallel_verify_action = {
	trigger = { NOT = { has_global_variable = nvm_par_verify_done } }
	effect = {
		if = {
			limit = { has_global_variable = nvm_par_skip_first }
			nvm_parallel_verify_impl = yes
			set_global_variable = nvm_par_verify_done
		} else = {
			set_global_variable = nvm_par_skip_first
		}
	}
}

nvm_parallel_clobber_action = {
	trigger = { NOT = { has_global_variable = nvm_par_clobber_done } }
	effect = {
		nvm_every_country_clobber_test = yes
		set_global_variable = nvm_par_clobber_done
	}
}
