# Runs after nvm_tutorial_initialize to verify the data model.

nvm_tutorial_run_tests = {

	# DOT-CHAIN SYNTAX VERIFICATION
	# Both dot-chaining and brace scoping work at runtime. Dot-chaining
	# produces parse-time "Unknown effect" warnings because the engine's
	# static analyzer can't resolve dynamic scopes. Brace scoping is
	# recommended to keep logs clean. These tests verify dot-chaining works.
	# ═══════════════════════════════════════════════════════════════════════

	# T0a: dot-chain through saved scope into NVM effect
	change_local_variable = { name = nvm_tut_total add = 1 }
	scope:nvm_tutorial_fra_horse_1.nvm_get_value = { key = flag:nvm_tutorial_speed }
	if = { limit = { NOT = { local_var:nvm_result >= 1 } }
		change_local_variable = { name = nvm_tut_failures add = 1 }
		error_log = "NVM_TUT T0a FAIL dot-chain scope.effect"
	}

	# T0b: dot-chain through country variable into NVM effect
	change_local_variable = { name = nvm_tut_total add = 1 }
	c:FRA.var:nvm_tutorial_stable.nvm_get_value = { key = flag:nvm_tutorial_total_wins }
	if = { limit = { NOT = { local_var:nvm_result >= 0 } }
		change_local_variable = { name = nvm_tut_failures add = 1 }
		error_log = "NVM_TUT T0b FAIL dot-chain country.var.effect"
	}

	change_local_variable = { name = nvm_tut_total add = 1 }

	if = { limit = { local_var:nvm_found = no }
		change_local_variable = { name = nvm_tut_failures add = 1 }
		error_log = "NVM_TUT T0c FAIL dot-chain set operation"
	}

	if = { limit = { NOT = { local_var:nvm_result = 5 } }
		change_local_variable = { name = nvm_tut_failures add = 1 }
		error_log = "NVM_TUT T25b FAIL weight value: expected 5 got [SCOPE.GetLocalVariable('nvm_result').GetValue|2]"
	}

	# ═══════════════════════════════════════════════════════════════════════
	# NVM_SET_MAP REPLACEMENT (overwrite existing keyed child)
	# ═══════════════════════════════════════════════════════════════════════

	# Horse 1 originally had warhorse maintenance, replace with packhorse
	change_local_variable = { name = nvm_tut_total add = 1 }
	scope:nvm_tutorial_fra_horse_1 = { nvm_set_map = { key = flag:nvm_tutorial_maintenance child = scope:nvm_tutorial_packhorse_maint } }
	scope:nvm_tutorial_fra_horse_1 = { nvm_get_map = { key = flag:nvm_tutorial_maintenance } }
	if = { limit = { local_var:nvm_found = no }
		change_local_variable = { name = nvm_tut_failures add = 1 }
		error_log = "NVM_TUT T26a FAIL replacement maintenance exists: expected found"
	}

	change_local_variable = { name = nvm_tut_total add = 1 }
	scope:nvm_cursor = { nvm_get_value = { key = goods:wheat } }
	if = { limit = { NOT = { local_var:nvm_result = 1 } }
		change_local_variable = { name = nvm_tut_failures add = 1 }
		error_log = "NVM_TUT T26b FAIL replaced maintenance wheat: expected 1 (packhorse) got [SCOPE.GetLocalVariable('nvm_result').GetValue|2]"
	}

	# ═══════════════════════════════════════════════════════════════════════
	# LIST POP (nvm_list_pop)
	# ═══════════════════════════════════════════════════════════════════════

	# France's horse 1 has a race history list with 1 entry (the race location)
	change_local_variable = { name = nvm_tut_total add = 1 }
	scope:nvm_tutorial_fra_horse_1 = { nvm_get_list = { key = flag:nvm_tutorial_race_history } }
	scope:nvm_cursor = { save_scope_as = nvm_tutorial_t_history }
	scope:nvm_tutorial_t_history = { nvm_list_pop = yes }
	if = { limit = { local_var:nvm_found = no }
		change_local_variable = { name = nvm_tut_failures add = 1 }
		error_log = "NVM_TUT T27a FAIL list pop returned item: expected found"
	}

	# List should now be empty
	change_local_variable = { name = nvm_tut_total add = 1 }
	scope:nvm_tutorial_t_history = {
		if = { limit = { NOT = { var:nvm_list_size = 0 } }
			change_local_variable = { name = nvm_tut_failures add = 1 }
			error_log = "NVM_TUT T27b FAIL list size after pop: expected 0 got [SCOPE.GetVariable('nvm_list_size').GetValue|0]"
		}
	}

	# Pop on empty list returns not found
	change_local_variable = { name = nvm_tut_total add = 1 }
	scope:nvm_tutorial_t_history = { nvm_list_pop = yes }
	if = { limit = { local_var:nvm_found = yes }
		change_local_variable = { name = nvm_tut_failures add = 1 }
		error_log = "NVM_TUT T27c FAIL pop empty list: expected not found"
	}

	# ═══════════════════════════════════════════════════════════════════════
	# SUMMARY
	# ═══════════════════════════════════════════════════════════════════════

	set_local_variable = { name = nvm_tut_passed value = local_var:nvm_tut_total }
	change_local_variable = { name = nvm_tut_passed subtract = local_var:nvm_tut_failures }
	error_log = "NVM_TUT SUMMARY: [SCOPE.GetLocalVariable('nvm_tut_passed').GetValue|0]/[SCOPE.GetLocalVariable('nvm_tut_total').GetValue|0] passed, [SCOPE.GetLocalVariable('nvm_tut_failures').GetValue|0] failures"

}
