nvm_test_functional_multi_parent = {
	nvm_new_map = yes
	scope:nvm_map = { save_temporary_scope_as = nvm_pa }
	nvm_new_map = yes
	scope:nvm_map = { save_temporary_scope_as = nvm_pb }

	scope:nvm_pa = { nvm_get_or_set_map = { key = goods:cloth } }
	scope:nvm_cursor = { save_temporary_scope_as = nvm_shared }

	scope:nvm_pb = {
		add_to_variable_map = { name = nvm_children key = goods:silk value = scope:nvm_shared }
	}
	scope:nvm_shared = {
		add_to_variable_list = { name = nvm_parents target = scope:nvm_pb }
	}

	set_local_variable = { name = v value = 88 }
	scope:nvm_shared = { nvm_set_value = { key = goods:paper value = local_var:v } }
	scope:nvm_pa = { nvm_get_map = { key = goods:cloth } }
	scope:nvm_cursor = { nvm_get_value = { key = goods:paper } }
	assert_if = { limit = { NOT = { local_var:nvm_result = 88 } } text = "T21a parent A reads shared: expected 88 got [SCOPE.GetLocalVariable('nvm_result').GetValue|2]" }
	scope:nvm_pb = { nvm_get_map = { key = goods:silk } }
	scope:nvm_cursor = { nvm_get_value = { key = goods:paper } }
	assert_if = { limit = { NOT = { local_var:nvm_result = 88 } } text = "T21b parent B reads shared: expected 88 got [SCOPE.GetLocalVariable('nvm_result').GetValue|2]" }

	scope:nvm_shared = { nvm_delete_map = yes }
	scope:nvm_pa = { nvm_has_map = { key = goods:cloth } }
	assert_if = { limit = { local_var:nvm_found = yes } text = "T21c parent A cleaned: expected not found got [SCOPE.GetLocalVariable('nvm_found').GetValue|0]" }
	scope:nvm_pb = { nvm_has_map = { key = goods:silk } }
	assert_if = { limit = { local_var:nvm_found = yes } text = "T21d parent B cleaned: expected not found got [SCOPE.GetLocalVariable('nvm_found').GetValue|0]" }

	nvm_new_map = yes
	scope:nvm_map = { save_temporary_scope_as = nvm_tree_root }
	scope:nvm_tree_root = { nvm_get_or_set_map = { key = goods:wheat } }
	scope:nvm_cursor = { save_temporary_scope_as = nvm_tree_child }
	scope:nvm_tree_child = { nvm_get_or_set_map = { key = goods:iron } }
	scope:nvm_cursor = { save_temporary_scope_as = nvm_tree_grandchild }
	set_local_variable = { name = v value = 123 }
	scope:nvm_tree_grandchild = { nvm_set_value = { key = goods:silk value = local_var:v } }

	scope:nvm_tree_root = { nvm_get_map = { key = goods:wheat } }
	scope:nvm_cursor = { nvm_get_map = { key = goods:iron } }
	scope:nvm_cursor = { nvm_get_value = { key = goods:silk } }
	assert_if = { limit = { NOT = { local_var:nvm_result = 123 } } text = "T22a pre-delete depth3: expected 123 got [SCOPE.GetLocalVariable('nvm_result').GetValue|2]" }

	scope:nvm_tree_root = { nvm_delete_tree = yes }
	scope:nvm_tree_grandchild = { nvm_has_value = { key = goods:silk } }
	assert_if = { limit = { local_var:nvm_found = yes } text = "T22b grandchild wiped: expected not found got [SCOPE.GetLocalVariable('nvm_found').GetValue|0]" }

	nvm_new_map = yes
	scope:nvm_map = { save_temporary_scope_as = nvm_recycle }
	set_local_variable = { name = v value = 999 }
	scope:nvm_recycle = { nvm_set_value = { key = goods:wheat value = local_var:v } }
	scope:nvm_recycle = { nvm_delete_map = yes }
	nvm_new_map = yes
	scope:nvm_map = { nvm_has_value = { key = goods:wheat } }
	assert_if = { limit = { local_var:nvm_found = yes } text = "T23 recycled clean: expected not found got [SCOPE.GetLocalVariable('nvm_found').GetValue|0]" }

	scope:nvm_pa = { nvm_delete_map = yes }
	scope:nvm_pb = { nvm_delete_map = yes }
}

nvm_test_functional_var_refs = {
	nvm_new_map = yes
	scope:nvm_map = { save_temporary_scope_as = nvm_var_test_dict }
	set_local_variable = { name = v value = 333 }
	scope:nvm_var_test_dict = { nvm_set_value = { key = goods:silk value = local_var:v } }

	scope:nvm_test_country = {
		set_variable = { name = nvm_my_dict value = scope:nvm_var_test_dict }
	}

	scope:nvm_test_country = {
		var:nvm_my_dict = { nvm_get_value = { key = goods:silk } }
	}
	assert_if = { limit = { NOT = { local_var:nvm_result = 333 } } text = "T24a var read: expected 333 got [SCOPE.GetLocalVariable('nvm_result').GetValue|2]" }

	set_local_variable = { name = v value = 444 }
	scope:nvm_test_country = {
		var:nvm_my_dict = { nvm_set_value = { key = goods:cloth value = local_var:v } }
	}
	scope:nvm_test_country = {
		var:nvm_my_dict = { nvm_get_value = { key = goods:cloth } }
	}
	assert_if = { limit = { NOT = { local_var:nvm_result = 444 } } text = "T24b var write/read: expected 444 got [SCOPE.GetLocalVariable('nvm_result').GetValue|2]" }

	scope:nvm_test_country = {
		var:nvm_my_dict = { nvm_get_or_set_map = { key = goods:wine } }
	}
	set_local_variable = { name = v value = 555 }
	scope:nvm_cursor = { nvm_set_value = { key = goods:tea value = local_var:v } }
	scope:nvm_test_country = {
		var:nvm_my_dict = { nvm_get_map = { key = goods:wine } }
	}
	scope:nvm_cursor = { nvm_get_value = { key = goods:tea } }
	assert_if = { limit = { NOT = { local_var:nvm_result = 555 } } text = "T24c var navigate child: expected 555 got [SCOPE.GetLocalVariable('nvm_result').GetValue|2]" }

	scope:nvm_test_country.var:nvm_my_dict = { nvm_get_value = { key = goods:silk } }
	assert_if = { limit = { NOT = { local_var:nvm_result = 333 } } text = "T24d scope.var block: expected 333 got [SCOPE.GetLocalVariable('nvm_result').GetValue|2]" }

	# scope.var.effect dot-chain syntax is NOT supported for effects.
	# scope:X.var:Y = { effect } works (T24d), but adding a third
	# segment for the effect name does not parse as a valid scope.

	scope:nvm_test_capital = {
		owner = {
			set_variable = { name = nvm_chain_test value = scope:nvm_var_test_dict }
		}
	}
	scope:nvm_test_capital.owner.var:nvm_chain_test = { nvm_get_value = { key = goods:silk } }
	assert_if = { limit = { NOT = { local_var:nvm_result = 333 } } text = "T24f deep chain: expected 333 got [SCOPE.GetLocalVariable('nvm_result').GetValue|2]" }

	set_local_variable = {
		name = r
		value = "scope:nvm_test_country.var:nvm_my_dict.variable_map(nvm_values|goods:silk)"
	}
	assert_if = { limit = { NOT = { local_var:r = 333 } } text = "T24g inline read: expected 333 got [SCOPE.GetLocalVariable('r').GetValue|0]" }

	scope:nvm_var_test_dict = { nvm_delete_tree = yes }
	scope:nvm_test_country = { remove_variable = nvm_my_dict }
	scope:nvm_test_capital = { owner = { remove_variable = nvm_chain_test } }
}

# DIAMOND DELETION (shared child survives parent flat-delete)
# ═══════════════════════════════════════════════════════════════════════════════

nvm_test_functional_diamond_delete = {
	nvm_new_map = yes
	scope:nvm_map = { save_temporary_scope_as = nvm_dia_pa }
	nvm_new_map = yes
	scope:nvm_map = { save_temporary_scope_as = nvm_dia_pb }

	# Create shared child under parent A
	scope:nvm_dia_pa = { nvm_get_or_set_map = { key = goods:wheat } }
	scope:nvm_cursor = { save_temporary_scope_as = nvm_dia_shared }

	# Link shared child to parent B
	scope:nvm_dia_pb = {
		add_to_variable_map = { name = nvm_children key = goods:iron value = scope:nvm_dia_shared }
	}
	scope:nvm_dia_shared = {
		add_to_variable_list = { name = nvm_parents target = scope:nvm_dia_pb }
	}

	set_local_variable = { name = v value = 42 }
	scope:nvm_dia_shared = { nvm_set_value = { key = goods:silk value = local_var:v } }

	# Flat-delete parent A: shared child should survive
	scope:nvm_dia_pa = { nvm_delete_map = yes }

	scope:nvm_dia_shared = { nvm_is_valid = yes }
	assert_if = { limit = { local_var:nvm_found = no } text = "diamond flat: shared still valid got [SCOPE.GetLocalVariable('nvm_found').GetValue|0]" }
	scope:nvm_dia_shared = { nvm_get_value = { key = goods:silk } }
	assert_if = { limit = { NOT = { local_var:nvm_result = 42 } } text = "diamond flat: shared data intact got [SCOPE.GetLocalVariable('nvm_result').GetValue|2]" }

	# Parent B still links to shared child
	scope:nvm_dia_pb = { nvm_get_map = { key = goods:iron } }
	assert_if = { limit = { local_var:nvm_found = no } text = "diamond flat: parent B link intact got [SCOPE.GetLocalVariable('nvm_found').GetValue|0]" }
	scope:nvm_cursor = {
		assert_if = { limit = { NOT = { this = scope:nvm_dia_shared } } text = "diamond flat: parent B same child" }
	}

	# Tree-delete parent B: should cascade to shared child
	scope:nvm_dia_pb = { nvm_delete_tree = yes }

	scope:nvm_dia_shared = { nvm_is_valid = yes }
	assert_if = { limit = { local_var:nvm_found = yes } text = "diamond tree: shared freed got [SCOPE.GetLocalVariable('nvm_found').GetValue|0]" }
}

	set_local_variable = { name = v value = 99 }

# ORPHANED PARENT REFERENCES (clear_maps/clear_lists then delete)
# ═══════════════════════════════════════════════════════════════════════════════

nvm_test_functional_orphan_cleanup = {
	nvm_new_map = yes
	scope:nvm_map = { save_temporary_scope_as = nvm_oc_parent }

	# Create a child map and a list
	scope:nvm_oc_parent = { nvm_get_or_set_map = { key = goods:wheat } }
	scope:nvm_cursor = { save_temporary_scope_as = nvm_oc_child }
	set_local_variable = { name = v value = 42 }
	scope:nvm_oc_child = { nvm_set_value = { key = goods:iron value = local_var:v } }

	scope:nvm_oc_parent = { nvm_get_or_set_list = { key = goods:silk } }
	scope:nvm_cursor = { save_temporary_scope_as = nvm_oc_list }
	scope:nvm_oc_list = { nvm_list_add = { value = goods:tea } }

	# clear_maps: removes children refs without unregistering from child's parents
	scope:nvm_oc_parent = { nvm_clear_maps = yes }
	scope:nvm_oc_parent = { nvm_has_map = { key = goods:wheat } }
	assert_if = { limit = { local_var:nvm_found = yes } text = "orphan clear_maps: ref removed got [SCOPE.GetLocalVariable('nvm_found').GetValue|0]" }

	# Child still functions despite stale parent ref
	scope:nvm_oc_child = { nvm_get_value = { key = goods:iron } }
	assert_if = { limit = { NOT = { local_var:nvm_result = 42 } } text = "orphan child functional: expected 42 got [SCOPE.GetLocalVariable('nvm_result').GetValue|2]" }

	# Deleting orphaned child should handle stale parent gracefully
	scope:nvm_oc_child = { nvm_delete_map = yes }
	scope:nvm_oc_child = { nvm_is_valid = yes }
	assert_if = { limit = { local_var:nvm_found = yes } text = "orphan child deleted: expected not valid got [SCOPE.GetLocalVariable('nvm_found').GetValue|0]" }

	# clear_lists: same pattern
	scope:nvm_oc_parent = { nvm_clear_lists = yes }
	scope:nvm_oc_parent = { nvm_has_list = { key = goods:silk } }
	assert_if = { limit = { local_var:nvm_found = yes } text = "orphan clear_lists: ref removed got [SCOPE.GetLocalVariable('nvm_found').GetValue|0]" }

	# Deleting orphaned list node should handle stale parent gracefully
	scope:nvm_oc_list = { nvm_delete_map = yes }
	scope:nvm_oc_list = { nvm_is_valid = yes }
	assert_if = { limit = { local_var:nvm_found = yes } text = "orphan list deleted: expected not valid got [SCOPE.GetLocalVariable('nvm_found').GetValue|0]" }

	# Parent still functions
	scope:nvm_oc_parent = { nvm_is_valid = yes }
	assert_if = { limit = { local_var:nvm_found = no } text = "orphan parent still valid: expected valid got [SCOPE.GetLocalVariable('nvm_found').GetValue|0]" }

	scope:nvm_oc_parent = { nvm_delete_map = yes }
}

	scope:nvm_im_test = { nvm_set_value = { key = goods:wheat value = 10 } }
	scope:nvm_im_test = { nvm_set_value = { key = goods:iron value = 20 } }

	set_local_variable = { name = iter_count value = 0 }

	assert_if = { limit = { NOT = { local_var:nvm_result = 42 } } text = "iter remove during: node functional after got [SCOPE.GetLocalVariable('nvm_result').GetValue|2]" }

	scope:nvm_im_test = { nvm_delete_map = yes }
}
