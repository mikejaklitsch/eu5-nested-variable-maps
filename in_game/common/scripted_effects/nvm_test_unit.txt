	assert_if = { limit = { NOT = { local_var:nvm_result = 75 } } text = "weighted after remove: iron intact got [SCOPE.GetLocalVariable('nvm_result').GetValue|2]" }

	# Clear wipes all weights
	scope:nvm_ws_set = { nvm_set_clear = yes }
	scope:nvm_ws_set = { nvm_set_get_weight = { value = goods:iron } }
	assert_if = { limit = { NOT = { local_var:nvm_result = 0 } } text = "weighted after clear: expected 0 got [SCOPE.GetLocalVariable('nvm_result').GetValue|2]" }

	scope:nvm_ws_parent = { nvm_delete_tree = yes }
}

nvm_test_unit_set_map = {
	nvm_new_map = yes
	scope:nvm_map = { save_temporary_scope_as = nvm_sm_parent }

	# Basic: adopt an existing node at a key
	nvm_new_map = yes
	scope:nvm_map = { save_temporary_scope_as = nvm_sm_child_a }
	set_local_variable = { name = v value = 42 }
	scope:nvm_sm_child_a = { nvm_set_value = { key = goods:iron value = local_var:v } }

	scope:nvm_sm_parent = { nvm_set_map = { key = goods:wheat child = scope:nvm_sm_child_a } }

	scope:nvm_sm_parent = { nvm_get_map = { key = goods:wheat } }
	assert_if = { limit = { local_var:nvm_found = no } text = "set_map basic: expected found got [SCOPE.GetLocalVariable('nvm_found').GetValue|0]" }
	scope:nvm_cursor = {
		assert_if = { limit = { NOT = { this = scope:nvm_sm_child_a } } text = "set_map basic: identity match" }
	}
	scope:nvm_cursor = { nvm_get_value = { key = goods:iron } }
	assert_if = { limit = { NOT = { local_var:nvm_result = 42 } } text = "set_map basic: data intact got [SCOPE.GetLocalVariable('nvm_result').GetValue|2]" }

	# has_map and check_key work on adopted children
	scope:nvm_sm_parent = { nvm_has_map = { key = goods:wheat } }
	assert_if = { limit = { local_var:nvm_found = no } text = "set_map has_map: expected found got [SCOPE.GetLocalVariable('nvm_found').GetValue|0]" }
	scope:nvm_sm_parent = { nvm_check_key = { key = goods:wheat } }
	assert_if = { limit = { local_var:nvm_has_map = no } text = "set_map check_key: expected has_map got [SCOPE.GetLocalVariable('nvm_has_map').GetValue|0]" }

	# Replace: adopt a different child at the same key
	nvm_new_map = yes
	scope:nvm_map = { save_temporary_scope_as = nvm_sm_child_b }
	set_local_variable = { name = v value = 77 }
	scope:nvm_sm_child_b = { nvm_set_value = { key = goods:silk value = local_var:v } }

	scope:nvm_sm_parent = { nvm_set_map = { key = goods:wheat child = scope:nvm_sm_child_b } }

	scope:nvm_sm_parent = { nvm_get_map = { key = goods:wheat } }
	scope:nvm_cursor = {
		assert_if = { limit = { NOT = { this = scope:nvm_sm_child_b } } text = "set_map replace: expected child_b" }
	}
	scope:nvm_cursor = { nvm_get_value = { key = goods:silk } }
	assert_if = { limit = { NOT = { local_var:nvm_result = 77 } } text = "set_map replace: child_b data got [SCOPE.GetLocalVariable('nvm_result').GetValue|2]" }

	# Replaced child still valid, data intact
	scope:nvm_sm_child_a = { nvm_is_valid = yes }
	assert_if = { limit = { local_var:nvm_found = no } text = "set_map replace: old child still valid got [SCOPE.GetLocalVariable('nvm_found').GetValue|0]" }
	scope:nvm_sm_child_a = { nvm_get_value = { key = goods:iron } }
	assert_if = { limit = { NOT = { local_var:nvm_result = 42 } } text = "set_map replace: old child data intact got [SCOPE.GetLocalVariable('nvm_result').GetValue|2]" }

	# Replace child created by nvm_get_or_set_map
	scope:nvm_sm_parent = { nvm_get_or_set_map = { key = goods:iron } }
	scope:nvm_cursor = { save_temporary_scope_as = nvm_sm_auto_child }
	set_local_variable = { name = v value = 33 }
	scope:nvm_sm_auto_child = { nvm_set_value = { key = goods:tea value = local_var:v } }

	nvm_new_map = yes
	scope:nvm_map = { save_temporary_scope_as = nvm_sm_child_c }
	scope:nvm_sm_parent = { nvm_set_map = { key = goods:iron child = scope:nvm_sm_child_c } }

	scope:nvm_sm_parent = { nvm_get_map = { key = goods:iron } }
	scope:nvm_cursor = {
		assert_if = { limit = { NOT = { this = scope:nvm_sm_child_c } } text = "set_map replace auto: expected child_c" }
	}
	scope:nvm_sm_auto_child = { nvm_is_valid = yes }
	assert_if = { limit = { local_var:nvm_found = no } text = "set_map replace auto: old auto child still valid got [SCOPE.GetLocalVariable('nvm_found').GetValue|0]" }

	# Self-assignment: adopt same child at same key (safe no-op on data)
	scope:nvm_sm_parent = { nvm_set_map = { key = goods:wheat child = scope:nvm_sm_child_b } }
	scope:nvm_sm_parent = { nvm_get_map = { key = goods:wheat } }
	scope:nvm_cursor = {
		assert_if = { limit = { NOT = { this = scope:nvm_sm_child_b } } text = "set_map self-assign: still child_b" }
	}

	# Adopt a node that already has a different parent
	nvm_new_map = yes
	scope:nvm_map = { save_temporary_scope_as = nvm_sm_other_parent }
	scope:nvm_sm_other_parent = { nvm_get_or_set_map = { key = goods:fur } }
	scope:nvm_cursor = { save_temporary_scope_as = nvm_sm_shared }
	set_local_variable = { name = v value = 88 }
	scope:nvm_sm_shared = { nvm_set_value = { key = goods:cloth value = local_var:v } }

	scope:nvm_sm_parent = { nvm_set_map = { key = goods:silk child = scope:nvm_sm_shared } }

	# Both parents can read the shared child
	scope:nvm_sm_parent = { nvm_get_map = { key = goods:silk } }
	scope:nvm_cursor = { nvm_get_value = { key = goods:cloth } }
	assert_if = { limit = { NOT = { local_var:nvm_result = 88 } } text = "set_map shared: parent reads 88 got [SCOPE.GetLocalVariable('nvm_result').GetValue|2]" }
	scope:nvm_sm_other_parent = { nvm_get_map = { key = goods:fur } }
	scope:nvm_cursor = { nvm_get_value = { key = goods:cloth } }
	assert_if = { limit = { NOT = { local_var:nvm_result = 88 } } text = "set_map shared: other parent reads 88 got [SCOPE.GetLocalVariable('nvm_result').GetValue|2]" }

	# delete_tree cascades through adopted children
	scope:nvm_sm_parent = { nvm_delete_tree = yes }
	scope:nvm_sm_child_b = { nvm_is_valid = yes }
	assert_if = { limit = { local_var:nvm_found = yes } text = "set_map tree delete: child_b freed got [SCOPE.GetLocalVariable('nvm_found').GetValue|0]" }
	scope:nvm_sm_child_c = { nvm_is_valid = yes }
	assert_if = { limit = { local_var:nvm_found = yes } text = "set_map tree delete: child_c freed got [SCOPE.GetLocalVariable('nvm_found').GetValue|0]" }

	# Clean up remaining nodes
	scope:nvm_sm_child_a = { nvm_delete_map = yes }
	scope:nvm_sm_auto_child = { nvm_delete_map = yes }
	scope:nvm_sm_other_parent = { nvm_delete_tree = yes }
}
