# ═══════════════════════════════════════════════════════════════════════════════
# NESTED MAP TEST SUITE
# ═══════════════════════════════════════════════════════════════════════════════
# Permanent test suite. Enable on_action dispatch in nvm_test.txt to run.
# Tests use assert_if: a failing assertion halts execution with a message.
# Filter output with: eu5-logs error -f NVM

nvm_run_tests_impl = {
	nvm_init_pool = yes

	# Count characters in the world (diagnostics)
	set_local_variable = { name = ct_characters value = 0 }
	every_country = {
		every_character = {
			change_local_variable = { name = ct_characters add = 1 }
		}
	}
	error_log = "NVM characters=[SCOPE.GetLocalVariable('ct_characters').GetValue|0]"

	# ═══════════════════════════════════════════════════════════════════════
	# SETUP
	# ═══════════════════════════════════════════════════════════════════════

	nvm_new_map = yes
	scope:nvm_map = { save_scope_as = nvm_test_root }

	random_country = {
		limit = { is_real_country = yes }
		save_scope_as = nvm_test_country
		capital = {
			save_scope_as = nvm_test_capital
			sub_continent = { save_scope_as = nvm_test_sc }
		}
	}

	# ═══════════════════════════════════════════════════════════════════════
	# VALUE OPERATIONS
	# ═══════════════════════════════════════════════════════════════════════

	# T0: bare literal value (no local_var)
	scope:nvm_test_root = { nvm_set_value = { key = goods:fur value = 13 } }
	scope:nvm_test_root = { nvm_get_value = { key = goods:fur } }
	assert_if = { limit = { NOT = { global_var:nvm_result = 13 } } text = "NVM T0 bare literal: expected 13" }
	scope:nvm_test_root = { nvm_remove_value = { key = goods:fur } }

	# T1: set and get a value
	set_local_variable = { name = v value = 42 }
	scope:nvm_test_root = { nvm_set_value = { key = goods:wheat value = local_var:v } }
	scope:nvm_test_root = { nvm_get_value = { key = goods:wheat } }
	assert_if = { limit = { NOT = { global_var:nvm_result = 42 } } text = "NVM T1 set/get: expected 42" }

	# T2: multiple values, first key survives
	set_local_variable = { name = v value = 77 }
	scope:nvm_test_root = { nvm_set_value = { key = goods:iron value = local_var:v } }
	scope:nvm_test_root = { nvm_get_value = { key = goods:iron } }
	assert_if = { limit = { NOT = { global_var:nvm_result = 77 } } text = "NVM T2a second key: expected 77" }
	scope:nvm_test_root = { nvm_get_value = { key = goods:wheat } }
	assert_if = { limit = { NOT = { global_var:nvm_result = 42 } } text = "NVM T2b first key intact: expected 42" }

	# T3: overwrite existing value
	set_local_variable = { name = v value = 99 }
	scope:nvm_test_root = { nvm_set_value = { key = goods:wheat value = local_var:v } }
	scope:nvm_test_root = { nvm_get_value = { key = goods:wheat } }
	assert_if = { limit = { NOT = { global_var:nvm_result = 99 } } text = "NVM T3 overwrite: expected 99" }

	# T4: add_to_value accumulates
	set_local_variable = { name = v value = 1 }
	scope:nvm_test_root = { nvm_change_value = { key = goods:wheat operation = add value = local_var:v } }
	scope:nvm_test_root = { nvm_get_value = { key = goods:wheat } }
	assert_if = { limit = { NOT = { global_var:nvm_result = 100 } } text = "NVM T4 add_to: expected 100" }

	# T5: add_to_value on missing key creates from 0
	set_local_variable = { name = v value = 5 }
	scope:nvm_test_root = { nvm_change_value = { key = goods:coffee operation = add value = local_var:v } }
	scope:nvm_test_root = { nvm_get_value = { key = goods:coffee } }
	assert_if = { limit = { NOT = { global_var:nvm_result = 5 } } text = "NVM T5 add_to missing key: expected 5" }

	# T6: multiply_value
	set_local_variable = { name = v value = 2 }
	scope:nvm_test_root = { nvm_change_value = { key = goods:wheat operation = multiply value = local_var:v } }
	scope:nvm_test_root = { nvm_get_value = { key = goods:wheat } }
	assert_if = { limit = { NOT = { global_var:nvm_result = 200 } } text = "NVM T6 multiply: expected 200" }

	# T7: swap_values
	scope:nvm_test_root = { nvm_swap_values = { key_a = goods:wheat key_b = goods:iron } }
	scope:nvm_test_root = { nvm_get_value = { key = goods:wheat } }
	assert_if = { limit = { NOT = { global_var:nvm_result = 77 } } text = "NVM T7a swap wheat: expected 77" }
	scope:nvm_test_root = { nvm_get_value = { key = goods:iron } }
	assert_if = { limit = { NOT = { global_var:nvm_result = 200 } } text = "NVM T7b swap iron: expected 200" }

	# T8: has_value true and false
	scope:nvm_test_root = { nvm_has_value = { key = goods:wheat } }
	assert_if = { limit = { global_var:nvm_found = no } text = "NVM T8a has wheat: expected found" }
	scope:nvm_test_root = { nvm_has_value = { key = goods:fur } }
	assert_if = { limit = { global_var:nvm_found = yes } text = "NVM T8b has fur: expected not found" }

	# T9: remove_value
	scope:nvm_test_root = { nvm_remove_value = { key = goods:iron } }
	scope:nvm_test_root = { nvm_has_value = { key = goods:iron } }
	assert_if = { limit = { global_var:nvm_found = yes } text = "NVM T9 removed iron: expected not found" }

	# T10: clear_values wipes all, children survive
	scope:nvm_test_root = { nvm_clear_values = yes }
	scope:nvm_test_root = { nvm_has_value = { key = goods:wheat } }
	assert_if = { limit = { global_var:nvm_found = yes } text = "NVM T10 clear_values: expected not found" }

	# ═══════════════════════════════════════════════════════════════════════
	# SUB-DICTIONARY NAVIGATION
	# ═══════════════════════════════════════════════════════════════════════

	# T11: add_map creates child, write/read on child, navigate back
	scope:nvm_test_root = { nvm_get_or_set_map = { key = scope:nvm_test_country } }
	set_local_variable = { name = v value = 55 }
	scope:nvm_cursor = { nvm_set_value = { key = goods:silk value = local_var:v } }
	scope:nvm_test_root = { nvm_get_map = { key = scope:nvm_test_country } }
	scope:nvm_cursor = { nvm_get_value = { key = goods:silk } }
	assert_if = { limit = { NOT = { global_var:nvm_result = 55 } } text = "NVM T11 depth2: expected 55" }

	# T12: add_map to existing child navigates without duplicating parent
	scope:nvm_test_root = { nvm_get_or_set_map = { key = scope:nvm_test_country } }
	scope:nvm_cursor = { nvm_get_value = { key = goods:silk } }
	assert_if = { limit = { NOT = { global_var:nvm_result = 55 } } text = "NVM T12 re-enter child: expected 55" }

	# T13: depth 3
	scope:nvm_test_root = { nvm_get_or_set_map = { key = scope:nvm_test_country } }
	scope:nvm_cursor = { nvm_get_or_set_map = { key = scope:nvm_test_sc } }
	set_local_variable = { name = v value = 7 }
	scope:nvm_cursor = { nvm_set_value = { key = goods:paper value = local_var:v } }
	scope:nvm_test_root = { nvm_get_map = { key = scope:nvm_test_country } }
	scope:nvm_cursor = { nvm_get_map = { key = scope:nvm_test_sc } }
	scope:nvm_cursor = { nvm_get_value = { key = goods:paper } }
	assert_if = { limit = { NOT = { global_var:nvm_result = 7 } } text = "NVM T13 depth3: expected 7" }

	# T14: multiple children, earlier child survives
	scope:nvm_test_root = { nvm_get_or_set_map = { key = goods:wine } }
	set_local_variable = { name = v value = 33 }
	scope:nvm_cursor = { nvm_set_value = { key = goods:cloth value = local_var:v } }
	scope:nvm_test_root = { nvm_get_map = { key = scope:nvm_test_country } }
	scope:nvm_cursor = { nvm_get_value = { key = goods:silk } }
	assert_if = { limit = { NOT = { global_var:nvm_result = 55 } } text = "NVM T14 first child intact: expected 55" }

	# T15: get_map returns found=no on missing key
	scope:nvm_test_root = { nvm_get_map = { key = goods:fur } }
	assert_if = { limit = { global_var:nvm_found = yes } text = "NVM T15 missing child: expected not found" }

	# T16: has_map true and false
	scope:nvm_test_root = { nvm_has_map = { key = scope:nvm_test_country } }
	assert_if = { limit = { global_var:nvm_found = no } text = "NVM T16a has map: expected found" }
	scope:nvm_test_root = { nvm_has_map = { key = goods:fur } }
	assert_if = { limit = { global_var:nvm_found = yes } text = "NVM T16b no map: expected not found" }

	# T17: remove_map unlinks child and unregisters from parent list
	scope:nvm_test_root = { nvm_remove_map = { key = goods:wine } }
	scope:nvm_test_root = { nvm_has_map = { key = goods:wine } }
	assert_if = { limit = { global_var:nvm_found = yes } text = "NVM T17 remove_map: expected not found" }

	# T17b: clear_maps removes all children references
	scope:nvm_test_root = { nvm_get_or_set_map = { key = goods:wine } }
	scope:nvm_test_root = { nvm_get_or_set_map = { key = goods:tea } }
	scope:nvm_test_root = { nvm_clear_maps = yes }
	scope:nvm_test_root = { nvm_has_map = { key = goods:wine } }
	assert_if = { limit = { global_var:nvm_found = yes } text = "NVM T17b clear_maps wine: expected not found" }
	scope:nvm_test_root = { nvm_has_map = { key = goods:tea } }
	assert_if = { limit = { global_var:nvm_found = yes } text = "NVM T17c clear_maps tea: expected not found" }
	# Rebuild the country child since clear_maps wiped it
	scope:nvm_test_root = { nvm_get_or_set_map = { key = scope:nvm_test_country } }
	set_local_variable = { name = v value = 55 }
	scope:nvm_cursor = { nvm_set_value = { key = goods:silk value = local_var:v } }

	# ═══════════════════════════════════════════════════════════════════════
	# SCOPE REFERENCES
	# ═══════════════════════════════════════════════════════════════════════

	# T18: store and retrieve a location reference
	scope:nvm_test_root = { nvm_set_scope = { key = goods:wine value = scope:nvm_test_capital } }
	scope:nvm_test_root = { nvm_get_scope = { key = goods:wine } }
	assert_if = { limit = { global_var:nvm_found = no } text = "NVM T18a scope found: expected found" }
	scope:nvm_scope = {
		assert_if = { limit = { NOT = { this = scope:nvm_test_capital } } text = "NVM T18b scope identity: expected capital" }
	}

	# T19: has_scope and remove_scope
	scope:nvm_test_root = { nvm_has_scope = { key = goods:wine } }
	assert_if = { limit = { global_var:nvm_found = no } text = "NVM T19a has scope: expected found" }
	scope:nvm_test_root = { nvm_remove_scope = { key = goods:wine } }
	scope:nvm_test_root = { nvm_has_scope = { key = goods:wine } }
	assert_if = { limit = { global_var:nvm_found = yes } text = "NVM T19b removed scope: expected not found" }

	# T20: clear_scopes
	scope:nvm_test_root = { nvm_set_scope = { key = goods:silk value = scope:nvm_test_capital } }
	scope:nvm_test_root = { nvm_set_scope = { key = goods:cloth value = scope:nvm_test_capital } }
	scope:nvm_test_root = { nvm_clear_scopes = yes }
	scope:nvm_test_root = { nvm_has_scope = { key = goods:silk } }
	assert_if = { limit = { global_var:nvm_found = yes } text = "NVM T20 clear_scopes: expected not found" }

	# ═══════════════════════════════════════════════════════════════════════
	# ARITHMETIC
	# ═══════════════════════════════════════════════════════════════════════

	# T20b: subtract_from_value
	set_local_variable = { name = v value = 50 }
	scope:nvm_test_root = { nvm_set_value = { key = goods:tea value = local_var:v } }
	set_local_variable = { name = v value = 15 }
	scope:nvm_test_root = { nvm_change_value = { key = goods:tea operation = subtract value = local_var:v } }
	scope:nvm_test_root = { nvm_get_value = { key = goods:tea } }
	assert_if = { limit = { NOT = { global_var:nvm_result = 35 } } text = "NVM T20b subtract: expected 35" }

	# T20c: divide_value
	set_local_variable = { name = v value = 5 }
	scope:nvm_test_root = { nvm_change_value = { key = goods:tea operation = divide value = local_var:v } }
	scope:nvm_test_root = { nvm_get_value = { key = goods:tea } }
	assert_if = { limit = { NOT = { global_var:nvm_result = 7 } } text = "NVM T20c divide: expected 7" }

	# ═══════════════════════════════════════════════════════════════════════
	# TYPE CHECKING
	# ═══════════════════════════════════════════════════════════════════════

	# T20d: has_scope true and false
	scope:nvm_test_root = { nvm_set_scope = { key = goods:fur value = scope:nvm_test_capital } }
	scope:nvm_test_root = { nvm_has_scope = { key = goods:fur } }
	assert_if = { limit = { global_var:nvm_found = no } text = "NVM T20d has scope: expected found" }
	scope:nvm_test_root = { nvm_has_scope = { key = goods:paper } }
	assert_if = { limit = { global_var:nvm_found = yes } text = "NVM T20e no scope: expected not found" }

	# T20f: check_key reports all three types at once
	# goods:tea has a value (T20c), goods:fur has a scope (T20d),
	# nvm_test_country has a map (T17b rebuild)
	scope:nvm_test_root = { nvm_check_key = { key = goods:tea } }
	assert_if = { limit = { global_var:nvm_has_value = no } text = "NVM T20f tea has_value: expected yes" }
	assert_if = { limit = { global_var:nvm_has_map = yes } text = "NVM T20g tea has_map: expected no" }
	scope:nvm_test_root = { nvm_check_key = { key = scope:nvm_test_country } }
	assert_if = { limit = { global_var:nvm_has_map = no } text = "NVM T20h country has_map: expected yes" }
	assert_if = { limit = { global_var:nvm_has_scope = yes } text = "NVM T20i country has_scope: expected no" }

	# Clean up test ref
	scope:nvm_test_root = { nvm_remove_scope = { key = goods:fur } }
	scope:nvm_test_root = { nvm_remove_value = { key = goods:tea } }

	# ═══════════════════════════════════════════════════════════════════════
	# MULTI-PARENT + DELETE
	# ═══════════════════════════════════════════════════════════════════════

	# T21: two parents share a child, delete_map cleans both
	nvm_new_map = yes
	scope:nvm_map = { save_scope_as = nvm_pa }
	nvm_new_map = yes
	scope:nvm_map = { save_scope_as = nvm_pb }

	# Create shared child via parent A
	scope:nvm_pa = { nvm_get_or_set_map = { key = goods:cloth } }
	scope:nvm_cursor = { save_scope_as = nvm_shared }

	# Parent B also links to the same child
	scope:nvm_pb = {
		add_to_variable_map = { name = nvm_children key = goods:silk value = scope:nvm_shared }
	}
	scope:nvm_shared = {
		add_to_variable_list = { name = nvm_parents target = scope:nvm_pb }
	}

	# Verify both parents reach it
	set_local_variable = { name = v value = 88 }
	scope:nvm_shared = { nvm_set_value = { key = goods:paper value = local_var:v } }
	scope:nvm_pa = { nvm_get_map = { key = goods:cloth } }
	scope:nvm_cursor = { nvm_get_value = { key = goods:paper } }
	assert_if = { limit = { NOT = { global_var:nvm_result = 88 } } text = "NVM T21a parent A reads shared: expected 88" }
	scope:nvm_pb = { nvm_get_map = { key = goods:silk } }
	scope:nvm_cursor = { nvm_get_value = { key = goods:paper } }
	assert_if = { limit = { NOT = { global_var:nvm_result = 88 } } text = "NVM T21b parent B reads shared: expected 88" }

	# Delete shared child, both parents cleaned
	scope:nvm_shared = { nvm_delete_map = yes }
	scope:nvm_pa = { nvm_has_map = { key = goods:cloth } }
	assert_if = { limit = { global_var:nvm_found = yes } text = "NVM T21c parent A cleaned: expected not found" }
	scope:nvm_pb = { nvm_has_map = { key = goods:silk } }
	assert_if = { limit = { global_var:nvm_found = yes } text = "NVM T21d parent B cleaned: expected not found" }

	# T22: delete_tree recursively frees a subtree
	nvm_new_map = yes
	scope:nvm_map = { save_scope_as = nvm_tree_root }
	scope:nvm_tree_root = { nvm_get_or_set_map = { key = goods:wheat } }
	scope:nvm_cursor = { save_scope_as = nvm_tree_child }
	scope:nvm_tree_child = { nvm_get_or_set_map = { key = goods:iron } }
	scope:nvm_cursor = { save_scope_as = nvm_tree_grandchild }
	set_local_variable = { name = v value = 123 }
	scope:nvm_tree_grandchild = { nvm_set_value = { key = goods:silk value = local_var:v } }

	# Verify depth before delete
	scope:nvm_tree_root = { nvm_get_map = { key = goods:wheat } }
	scope:nvm_cursor = { nvm_get_map = { key = goods:iron } }
	scope:nvm_cursor = { nvm_get_value = { key = goods:silk } }
	assert_if = { limit = { NOT = { global_var:nvm_result = 123 } } text = "NVM T22a pre-delete depth3: expected 123" }

	# Delete tree from root
	scope:nvm_tree_root = { nvm_delete_tree = yes }

	# Grandchild should be wiped (variable map gone)
	scope:nvm_tree_grandchild = { nvm_has_value = { key = goods:silk } }
	assert_if = { limit = { global_var:nvm_found = yes } text = "NVM T22b grandchild wiped: expected not found" }

	# T23: pool recycling, allocate + delete + reallocate gives clean node
	nvm_new_map = yes
	scope:nvm_map = { save_scope_as = nvm_recycle }
	set_local_variable = { name = v value = 999 }
	scope:nvm_recycle = { nvm_set_value = { key = goods:wheat value = local_var:v } }
	scope:nvm_recycle = { nvm_delete_map = yes }

	# Reallocate; new_map should wipe stale data
	nvm_new_map = yes
	scope:nvm_map = { nvm_has_value = { key = goods:wheat } }
	assert_if = { limit = { global_var:nvm_found = yes } text = "NVM T23 recycled clean: expected not found" }

	# ═══════════════════════════════════════════════════════════════════════
	# VARIABLE-STORED DICTIONARY REFERENCE
	# Tests accessing a dictionary via var:X on a game scope.
	# ═══════════════════════════════════════════════════════════════════════

	# T24: store dict reference as variable on a country, access via var:
	nvm_new_map = yes
	scope:nvm_map = { save_scope_as = nvm_var_test_dict }
	set_local_variable = { name = v value = 333 }
	scope:nvm_var_test_dict = { nvm_set_value = { key = goods:silk value = local_var:v } }

	# Store the dictionary reference on a country as a plain variable
	scope:nvm_test_country = {
		set_variable = { name = nvm_my_dict value = scope:nvm_var_test_dict }
	}

	# Access it via var: on the country
	scope:nvm_test_country = {
		var:nvm_my_dict = { nvm_get_value = { key = goods:silk } }
	}
	assert_if = { limit = { NOT = { global_var:nvm_result = 333 } } text = "NVM T24a var read: expected 333" }

	# Write via var: and read back
	set_local_variable = { name = v value = 444 }
	scope:nvm_test_country = {
		var:nvm_my_dict = { nvm_set_value = { key = goods:cloth value = local_var:v } }
	}
	scope:nvm_test_country = {
		var:nvm_my_dict = { nvm_get_value = { key = goods:cloth } }
	}
	assert_if = { limit = { NOT = { global_var:nvm_result = 444 } } text = "NVM T24b var write/read: expected 444" }

	# Navigate into a child via var:
	scope:nvm_test_country = {
		var:nvm_my_dict = { nvm_get_or_set_map = { key = goods:wine } }
	}
	set_local_variable = { name = v value = 555 }
	scope:nvm_cursor = { nvm_set_value = { key = goods:tea value = local_var:v } }
	scope:nvm_test_country = {
		var:nvm_my_dict = { nvm_get_map = { key = goods:wine } }
	}
	scope:nvm_cursor = { nvm_get_value = { key = goods:tea } }
	assert_if = { limit = { NOT = { global_var:nvm_result = 555 } } text = "NVM T24c var navigate child: expected 555" }

	# T24d: scope block entry via var
	scope:nvm_test_country.var:nvm_my_dict = { nvm_get_value = { key = goods:silk } }
	assert_if = { limit = { NOT = { global_var:nvm_result = 333 } } text = "NVM T24d scope.var block: expected 333" }

	# T24e: dot chain directly into scripted effect
	scope:nvm_test_country.var:nvm_my_dict.nvm_get_value = { key = goods:silk }
	assert_if = { limit = { NOT = { global_var:nvm_result = 333 } } text = "NVM T24e dot chain effect: expected 333" }

	# T24f: dot chain from scope traversal (capital.owner.var)
	scope:nvm_test_capital = {
		owner = {
			set_variable = { name = nvm_chain_test value = scope:nvm_var_test_dict }
		}
	}
	scope:nvm_test_capital.owner.var:nvm_chain_test = { nvm_get_value = { key = goods:silk } }
	assert_if = { limit = { NOT = { global_var:nvm_result = 333 } } text = "NVM T24f deep chain: expected 333" }

	# T24g: inline read via dot chain in a value expression
	set_local_variable = {
		name = r
		value = "scope:nvm_test_country.var:nvm_my_dict.variable_map(nvm_values|goods:silk)"
	}
	assert_if = { limit = { NOT = { local_var:r = 333 } } text = "NVM T24g inline read: expected 333" }

	# Cleanup
	scope:nvm_var_test_dict = { nvm_delete_tree = yes }
	scope:nvm_test_country = { remove_variable = nvm_my_dict }
	scope:nvm_test_capital = { owner = { remove_variable = nvm_chain_test } }

	# ═══════════════════════════════════════════════════════════════════════
	# LISTS
	# ═══════════════════════════════════════════════════════════════════════

	# T30: create list and add items
	scope:nvm_test_root = { nvm_get_or_set_list = { key = goods:fur } }
	scope:nvm_cursor = { save_scope_as = nvm_test_list }
	scope:nvm_test_list = { nvm_list_add = { value = goods:wheat } }
	scope:nvm_test_list = { nvm_list_add = { value = goods:iron } }
	scope:nvm_test_list = { nvm_list_add = { value = goods:silk } }
	scope:nvm_test_list = {
		assert_if = { limit = { NOT = { var:nvm_list_size = 3 } } text = "NVM T30 list size: expected 3" }
	}

	assert_if = { limit = { global_var:nvm_has_map = no } text = "NVM T89b map survives: expected yes" }
	assert_if = { limit = { global_var:nvm_has_list = no } text = "NVM T89c list survives: expected yes" }

	# T90: removing map preserves list
	scope:nvm_adv = { nvm_remove_map = { key = goods:wheat } }
	scope:nvm_adv = { nvm_check_key = { key = goods:wheat } }
	assert_if = { limit = { global_var:nvm_has_map = yes } text = "NVM T90a map gone: expected no" }
	assert_if = { limit = { global_var:nvm_has_list = no } text = "NVM T90b list survives: expected yes" }

	scope:nvm_adv = { nvm_delete_tree = yes }

	# ═══════════════════════════════════════════════════════════════════════
	# SHARED LIST/SET PARENTS
	# ═══════════════════════════════════════════════════════════════════════

	# T91: list shared between two parents, delete cleans both
	nvm_new_map = yes
	scope:nvm_map = { save_scope_as = nvm_sp_a }
	nvm_new_map = yes
	scope:nvm_map = { save_scope_as = nvm_sp_b }

	scope:nvm_sp_a = { nvm_get_or_set_list = { key = goods:wheat } }
	scope:nvm_cursor = { save_scope_as = nvm_shared_list }
	scope:nvm_shared_list = { nvm_list_add = { value = goods:silk } }

	# Link same list to parent B manually
	scope:nvm_sp_b = {
		add_to_variable_map = { name = nvm_lists key = goods:iron value = scope:nvm_shared_list }
	}
	scope:nvm_shared_list = {
		add_to_variable_list = { name = nvm_parents target = scope:nvm_sp_b }
	}

	# Verify both parents reach the list
	scope:nvm_sp_a = { nvm_get_list = { key = goods:wheat } }
	scope:nvm_cursor = { nvm_list_has = { value = goods:silk } }
	assert_if = { limit = { global_var:nvm_found = no } text = "NVM T91a parent A reads list: expected found" }
	scope:nvm_sp_b = { nvm_get_list = { key = goods:iron } }
	scope:nvm_cursor = { nvm_list_has = { value = goods:silk } }
	assert_if = { limit = { global_var:nvm_found = no } text = "NVM T91b parent B reads list: expected found" }

	# Delete shared list, both parents should be cleaned
	scope:nvm_shared_list = { nvm_delete_map = yes }
	scope:nvm_sp_a = { nvm_has_list = { key = goods:wheat } }
	assert_if = { limit = { global_var:nvm_found = yes } text = "NVM T91c parent A cleaned: expected not found" }
	scope:nvm_sp_b = { nvm_has_list = { key = goods:iron } }
	assert_if = { limit = { global_var:nvm_found = yes } text = "NVM T91d parent B cleaned: expected not found" }

	scope:nvm_sp_a = { nvm_delete_map = yes }
	scope:nvm_sp_b = { nvm_delete_map = yes }

	# T92: set shared between two parents, delete cleans both
	nvm_new_map = yes
	scope:nvm_map = { save_scope_as = nvm_sp_a }
	nvm_new_map = yes
	scope:nvm_map = { save_scope_as = nvm_sp_b }

	scope:nvm_sp_a = { nvm_get_or_set_set = { key = goods:cloth } }
	scope:nvm_cursor = { save_scope_as = nvm_shared_set }
	scope:nvm_shared_set = { nvm_set_add = { value = goods:tea } }

	# Link same set to parent B manually
	scope:nvm_sp_b = {
		add_to_variable_map = { name = nvm_lists key = goods:silk value = scope:nvm_shared_set }
	}
	scope:nvm_shared_set = {
		add_to_variable_list = { name = nvm_parents target = scope:nvm_sp_b }
	}

	# Verify both parents reach the set
	scope:nvm_sp_a = { nvm_get_set = { key = goods:cloth } }
	scope:nvm_cursor = { nvm_set_has = { value = goods:tea } }
	assert_if = { limit = { global_var:nvm_found = no } text = "NVM T92a parent A reads set: expected found" }
	scope:nvm_sp_b = { nvm_get_set = { key = goods:silk } }
	scope:nvm_cursor = { nvm_set_has = { value = goods:tea } }
	assert_if = { limit = { global_var:nvm_found = no } text = "NVM T92b parent B reads set: expected found" }

	# Delete shared set, both parents should be cleaned
	scope:nvm_shared_set = { nvm_delete_map = yes }
	scope:nvm_sp_a = { nvm_has_list = { key = goods:cloth } }
	assert_if = { limit = { global_var:nvm_found = yes } text = "NVM T92c parent A cleaned: expected not found" }
	scope:nvm_sp_b = { nvm_has_list = { key = goods:silk } }
	assert_if = { limit = { global_var:nvm_found = yes } text = "NVM T92d parent B cleaned: expected not found" }

	scope:nvm_sp_a = { nvm_delete_map = yes }
	scope:nvm_sp_b = { nvm_delete_map = yes }

	# ═══════════════════════════════════════════════════════════════════════
	# DEAD CHARACTER STORAGE
	# ═══════════════════════════════════════════════════════════════════════
	# Tests whether dead characters retain variable map data.
	# If yes, characters become an expandable storage pool with no ceiling.

	# T100: read dead character from a PREVIOUS test run (months later)
	if = {
		limit = { has_global_variable = nvm_dead_char_persist }
		global_var:nvm_dead_char_persist = {
			assert_if = {
				limit = { NOT = { is_key_in_variable_map = { name = nvm_char_test target = goods:wheat } } }
				text = "NVM T100a dead char months later map: expected key"
			}
		}
		global_var:nvm_dead_char_persist = {
			set_global_variable = { name = nvm_result value = "variable_map(nvm_char_test|goods:wheat)" }
		}
		assert_if = { limit = { NOT = { global_var:nvm_result = 42 } } text = "NVM T100b dead char months later read: expected 42" }
		set_local_variable = { name = r value = global_var:nvm_dead_char_persist.var:nvm_char_flag }
		assert_if = { limit = { NOT = { local_var:r = 777 } } text = "NVM T100c dead char months later var: expected 777" }
	} else = {
		error_log = "NVM T100 no prior dead char (first run, skip)"
	}

	# T98: create character, write data, kill, read back
	random_country = {
		limit = { is_player = no }
		create_character = { save_scope_as = nvm_test_char }
	}
	if = {
		limit = { NOT = { exists = scope:nvm_test_char } }
		random_country = {
			create_character = { save_scope_as = nvm_test_char }
		}
	}
	scope:nvm_test_char = {
		add_to_variable_map = { name = nvm_char_test key = goods:wheat value = 42 }
		add_to_variable_map = { name = nvm_char_test key = goods:iron value = 99 }
		set_variable = { name = nvm_char_flag value = 777 }
	}
	kill_character_silently = scope:nvm_test_char

	# Read variable map after death
	scope:nvm_test_char = {
		assert_if = {
			limit = { NOT = { is_key_in_variable_map = { name = nvm_char_test target = goods:wheat } } }
			text = "NVM T98a dead char map exists: expected key"
		}
	}
	scope:nvm_test_char = {
		set_global_variable = { name = nvm_result value = "variable_map(nvm_char_test|goods:wheat)" }
	}
	assert_if = { limit = { NOT = { global_var:nvm_result = 42 } } text = "NVM T98b dead char map read: expected 42" }

	# Read regular variable after death
	set_local_variable = { name = r value = scope:nvm_test_char.var:nvm_char_flag }
	assert_if = { limit = { NOT = { local_var:r = 777 } } text = "NVM T98c dead char var: expected 777" }

	# T99: write NEW data to dead character
	scope:nvm_test_char = {
		add_to_variable_map = { name = nvm_char_test key = goods:silk value = 55 }
	}
	scope:nvm_test_char = {
		assert_if = {
			limit = { NOT = { is_key_in_variable_map = { name = nvm_char_test target = goods:silk } } }
			text = "NVM T99 dead char post-mortem write: expected key"
		}
	}

	# Save for T100 to verify on next run (months later)
	set_global_variable = { name = nvm_dead_char_persist value = scope:nvm_test_char }

	# ═══════════════════════════════════════════════════════════════════════
	# FLAG COMPATIBILITY
	# ═══════════════════════════════════════════════════════════════════════
	# Tests whether flag: (localization key references) work in variable
	# maps and variable lists. Determines feasibility of string-based
	# mod registration for the multi-mod pool budget system.

	# T93: flag as scoped variable map key (on pool node)
	nvm_new_map = yes
	scope:nvm_map = { save_scope_as = nvm_flag_node }
	set_local_variable = { name = v value = 50 }
	scope:nvm_flag_node = {
		add_to_variable_map = { name = nvm_flag_map key = flag:nvm_test_alpha value = local_var:v }
	}
	scope:nvm_flag_node = {
		assert_if = {
			limit = { NOT = { is_key_in_variable_map = { name = nvm_flag_map target = flag:nvm_test_alpha } } }
			text = "NVM T93a flag scoped map key: expected found"
		}
	}
	scope:nvm_flag_node = {
		set_global_variable = { name = nvm_result value = "variable_map(nvm_flag_map|flag:nvm_test_alpha)" }
	}
	assert_if = { limit = { NOT = { global_var:nvm_result = 50 } } text = "NVM T93b flag scoped map read: expected 50" }

	# T94: flag as global variable map key
	set_local_variable = { name = v value = 75 }
	add_to_global_variable_map = { name = nvm_flag_gmap key = flag:nvm_test_beta value = local_var:v }
	assert_if = {
		limit = { NOT = { is_key_in_global_variable_map = { name = nvm_flag_gmap target = flag:nvm_test_beta } } }
		text = "NVM T94a flag global map key: expected found"
	}
	set_local_variable = { name = r value = "global_variable_map(nvm_flag_gmap|flag:nvm_test_beta)" }
	assert_if = { limit = { NOT = { local_var:r = 75 } } text = "NVM T94b flag global map read: expected 75" }

	# T95: flag as variable map value (store and identity-compare)
	add_to_global_variable_map = { name = nvm_flag_val key = goods:wheat value = flag:nvm_test_gamma }
	assert_if = {
		limit = { NOT = { is_key_in_global_variable_map = { name = nvm_flag_val target = goods:wheat } } }
		text = "NVM T95a flag as map value stored: expected found"
	}
	"global_variable_map(nvm_flag_val|goods:wheat)" = {
		assert_if = { limit = { NOT = { this = flag:nvm_test_gamma } } text = "NVM T95b flag value identity: expected gamma" }
	}

	# T96: flag in global variable list
	add_to_global_variable_list = { name = nvm_flag_list target = flag:nvm_test_delta }
	assert_if = {
		limit = { NOT = { is_target_in_global_variable_list = { name = nvm_flag_list target = flag:nvm_test_delta } } }
		text = "NVM T96a flag in list: expected found"
	}
	# Iterate and identity-compare
	set_global_variable = { name = nvm_found value = no }
	every_in_global_list = {
		variable = nvm_flag_list
		if = { limit = { this = flag:nvm_test_delta }
			set_global_variable = { name = nvm_found value = yes }
		}
	}
	assert_if = { limit = { global_var:nvm_found = no } text = "NVM T96b flag list iterate: expected found" }

	# T97: flag stored as variable, read back with compare
	scope:nvm_flag_node = {
		set_variable = { name = nvm_owner value = flag:nvm_test_alpha }
	}
	scope:nvm_flag_node = {
		assert_if = { limit = { NOT = { var:nvm_owner = flag:nvm_test_alpha } } text = "NVM T97 flag var compare: expected match" }
	}

	# Cleanup
	scope:nvm_flag_node = { nvm_delete_map = yes }
	clear_global_variable_map = nvm_flag_gmap
	clear_global_variable_map = nvm_flag_val
	clear_global_variable_list = nvm_flag_list

	# ═══════════════════════════════════════════════════════════════════════
	# INTEGRITY
	# ═══════════════════════════════════════════════════════════════════════

	# T27: original test root survived everything (values were cleared in T10)
	scope:nvm_test_root = { nvm_get_map = { key = scope:nvm_test_country } }
	scope:nvm_cursor = { nvm_get_value = { key = goods:silk } }
	assert_if = { limit = { NOT = { global_var:nvm_result = 55 } } text = "NVM T27 root child intact: expected 55" }

	error_log = "NVM tests complete"
}
